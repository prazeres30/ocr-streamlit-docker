FROM python:3.10-slim# Instalar dependências do sistemaRUN apt-get update && apt-get install -y \    tesseract-ocr \    poppler-utils \    libgl1 \    libglib2.0-0 \    && rm -rf /var/lib/apt/lists/*# Instalar dependências PythonCOPY requirements.txt .RUN pip install --no-cache-dir -r requirements.txt# Copiar o appCOPY . /appWORKDIR /appEXPOSE 8501CMD ["streamlit", "run", "app.py", "--server.port=8501", "--server.address=0.0.0.0"]
